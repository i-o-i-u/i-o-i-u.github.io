**الجواب:**

أقول: إذا أردتَّ أمرًا فجمِّعْ له جَرَاميزَك، وشُدَّ له حَيَازِيمَك، واقْرَعْ لهُ ظَنابِيبَك، وأَوِّبْ ولا تُغَوِّرْ، وأَسْئِدْ ولا تُعرِّسْ، ولا تذخَرْ طاقةً ولا تألُ جهدًا، فإذا استنفَدتَّ وُسْعَك، وأردَيْتَ نِضْوَك، فعندئذٍ فعَداك عَتْبٌ وخَلاك ذَمٌّ (ومُبْلِغُ نَفْسٍ عُذْرَها مِثْلُ مُنجِحِ!). أمَّا أن تُخلِدَ إلى الدَّعة، وتركنَ إلى الخَفْض، لم تُعَنِّ نفسَك في تحصيل الوسائل ودَرْك الغايات، ثم تنتظرَ أن يجريَ على لسانك الشعر البليغ، والكلام العالي، وأن يتهادى إلى قلَمك وَشْي البُلغاء وتطريز مُبرِّزي الأدباء، ويستقيمَ بيانُك على الطريقة! فلا!

<script>
const TASHKEEL = /[\u064B-\u065F\u0670]/;
const NO_CONNECT_AFTER = new Set(['ا','أ','إ','آ','ٱ','و','ؤ','ز','ذ','د','ر','ى','ة']);
const PUNCT = new Set(['،','؛','؟','!','.',',',':','-','–','—','«','»','"','"','(',')','/']);

function tokenize(word) {
  const tokens = [];
  for (let i = 0; i < word.length; i++) {
    if (TASHKEEL.test(word[i])) {
      if (tokens.length) tokens[tokens.length - 1].marks += word[i];
    } else {
      tokens.push({ base: word[i], marks: '' });
    }
  }
  return tokens;
}

function tokensToString(tokens) {
  return tokens.map(t => t.base + t.marks).join('');
}

function canConnectAfter(base) {
  return !NO_CONNECT_AFTER.has(base) && !PUNCT.has(base) && /[\u0600-\u06FF]/.test(base);
}

function getBestInsertPos(tokens) {
  let lastValid = tokens.length - 1;
  while (lastValid > 0 && PUNCT.has(tokens[lastValid].base)) lastValid--;
  for (let i = lastValid - 1; i >= 1; i--) {
    if (PUNCT.has(tokens[i + 1]?.base)) continue;
    const nextBase = tokens[i + 1]?.base;
    if (tokens[i].base === 'ل' && nextBase && 'اأإآٱ'.includes(nextBase)) continue;
    if (canConnectAfter(tokens[i].base)) return i + 1;
  }
  return -1;
}

function measureText(text, font) {
  const c = measureText._canvas || (measureText._canvas = document.createElement('canvas'));
  const ctx = c.getContext('2d');
  ctx.font = font;
  return ctx.measureText(text).width;
}

function stretchCell(cell, font) {
  const available = cell.clientWidth
    - parseFloat(getComputedStyle(cell).paddingLeft)
    - parseFloat(getComputedStyle(cell).paddingRight);
  if (!cell.dataset.original) cell.dataset.original = cell.textContent.trim();
  const words = cell.dataset.original.split(/\s+/).map(w => ({ tokens: tokenize(w), tatweels: 0 }));
  for (let round = 0; round < 80; round++) {
    const currentText = words.map(w => tokensToString(w.tokens)).join(' ');
    const gap = available - measureText(currentText, font);
    if (gap <= 0.5) break;
    const candidates = words
      .map((w, i) => ({ i, pos: getBestInsertPos(w.tokens), tatweels: w.tatweels, len: w.tokens.length }))
      .filter(c => c.pos !== -1)
      .sort((a, b) => a.tatweels - b.tatweels || b.len - a.len);
    if (!candidates.length) break;
    const best = candidates[0];
    words[best.i].tokens.splice(best.pos, 0, { base: 'ـ', marks: '' });
    words[best.i].tatweels++;
  }
  cell.textContent = words.map(w => tokensToString(w.tokens)).join(' ');
}

function stretchAll() {
  document.querySelectorAll('.poetry-table').forEach(table => {
    table.querySelectorAll('.shatr').forEach(cell => {
      if (cell.dataset.original) cell.textContent = cell.dataset.original;
    });
    const sample = table.querySelector('.shatr');
    if (!sample) return;
    const cs = getComputedStyle(sample);
    const font = `${cs.fontStyle} ${cs.fontWeight} ${cs.fontSize}/${cs.lineHeight} ${cs.fontFamily}`;
    table.querySelectorAll('.shatr').forEach(cell => stretchCell(cell, font));
  });
}

if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', stretchAll);
} else {
  stretchAll();
}
window.addEventListener('resize', stretchAll);
</script>
